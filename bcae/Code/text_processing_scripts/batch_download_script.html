def split_into_batches(input_filename, batch_size, output_folder):
    with open(input_filename, "r", encoding="utf-8") as input_file:
        lines = input_file.readlines()
    
    num_batches = (len(lines) + batch_size - 1) // batch_size
    
    for batch_num in range(num_batches):
        start_idx = batch_num * batch_size
        end_idx = min((batch_num + 1) * batch_size, len(lines))
        batch = lines[start_idx:end_idx]
        
        output_filename = f"batch_{batch_num + 1}.txt"
        output_path = os.path.join(output_folder, output_filename)
        
        with open(output_path, "w", encoding="utf-8") as output_file:
            output_file.writelines(batch)
        
        print(f"Saved batch {batch_num + 1} to '{output_filename}'")

import os
import requests

def download_file(url, output_folder):
    response = requests.get(url)
    
    if response.status_code == 200:
        filename = url.split("/")[-1]
        output_path = os.path.join(output_folder, filename)
        
        with open(output_path, "wb") as output_file:
            output_file.write(response.content)
        
        print(f"Downloaded '{filename}' to '{output_folder}'")
    else:
        print(f"Failed to download '{url}'")

def process_txt_file(txt_filename, output_folder):
    with open(txt_filename, "r", encoding="utf-8") as txt_file:
        urls = txt_file.readlines()
    
    batch_name = os.path.splitext(os.path.basename(txt_filename))[0]
    batch_folder = os.path.join(output_folder, batch_name)
    
    if not os.path.exists(batch_folder):
        os.makedirs(batch_folder)
    
    for url in urls:
        url = url.strip()
        download_file(url, batch_folder)

def main():
    input_folder = "/Users/dmitrijmazanik/Downloads/links_batches"  # Change to the folder containing your text files
    output_folder = "/Users/dmitrijmazanik/Downloads/中国工程院院刊"  # Change to the folder where you want to store downloaded files
    
    if not os.path.exists(output_folder):
        os.makedirs(output_folder)
    
    for txt_filename in os.listdir(input_folder):
        if txt_filename.endswith(".txt"):
            txt_path = os.path.join(input_folder, txt_filename)
            process_txt_file(txt_path, output_folder)

if __name__ == "__main__":
    main()
